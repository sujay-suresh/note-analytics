Extraction Scripts

Run these in the browser console on the corresponding note.com page.

1. Notifications

Open your notifications page (note.com/notifications), scroll to load all, then run: 

// Note.com Analytics -Notifications Extractor
// 1. Open https://note.com/notifications
// 2. Scroll down to load ALL notifications you want
// 3. Paste this entire script into the browser console
// 4. JSON is copied to clipboard -paste into the app

(async () => {
  const rows = document.querySelectorAll(
    '[class*="NotificationItem"], [class*="notification"] li, ' +
    'a[href*="/n/"]'
  );

  // Approach 1: Try __NEXT_DATA__ for structured data
  const nd = document.getElementById('__NEXT_DATA__');
  let notifications = [];

  if (nd) {
    try {
      const data = JSON.parse(nd.textContent);
      const pages = data?.props?.pageProps;
      const notifs = pages?.notifications || pages?.items || [];
      if (notifs.length > 0) {
        for (const n of notifs) {
          const kind = (n.type || '').includes('follow') ? 'follow' : 'like';
          const users = Array.isArray(n.users) ? n.users : (n.user ? [n.user] : []);
          notifications.push({
            kind,
            who: users.map(u => u.nickname || u.name || '').join('; '),
            who_url: users.map(u => 'https://note.com/' + (u.urlname || '')).join('; '),
            noticed_at_iso: n.createdAt || n.timestamp || '',
            noticed_at_jst: '',
            article_title: n.note?.name || n.note?.title || (kind === 'follow' ? '\u2014' : ''),
            article_url: n.note?.key ? 'https://note.com/' + (n.note.user?.urlname || '') + '/n/' + n.note.key : '\u2014',
            liker_profile: kind === 'follow' ? ('https://note.com/' + (users[0]?.urlname || '')) : '\u2014',
          });
        }
      }
    } catch (e) { console.warn('__NEXT_DATA__ parse failed, falling back to DOM', e); }
  }

  // Approach 2: DOM fallback -parse notification list items
  if (notifications.length === 0) {
    const items = document.querySelectorAll('[class*="Notification"] > div, [role="listitem"]');
    for (const item of items) {
      const links = item.querySelectorAll('a[href*="note.com/"]');
      const timeEl = item.querySelector('time, [datetime], [class*="time"], [class*="date"]');
      const timestamp = timeEl?.getAttribute('datetime') || timeEl?.textContent?.trim() || '';

      const userLinks = [];
      const articleLinks = [];
      for (const link of links) {
        const href = link.href || '';
        if (href.includes('/n/')) articleLinks.push(link);
        else if (href.match(/note\.com\/[^/]+\/?$/)) userLinks.push(link);
      }

      const text = item.textContent || '';
      const isFollow = text.includes('\u30d5\u30a9\u30ed\u30fc') || text.includes('follow');
      const kind = isFollow ? 'follow' : 'like';

      notifications.push({
        kind,
        who: userLinks.map(l => l.textContent?.trim() || '').join('; '),
        who_url: userLinks.map(l => l.href).join('; '),
        noticed_at_iso: timestamp,
        noticed_at_jst: '',
        article_title: articleLinks[0]?.textContent?.trim() || (isFollow ? '\u2014' : ''),
        article_url: articleLinks[0]?.href?.replace('?from=notice', '') || '\u2014',
        liker_profile: isFollow ? (userLinks[0]?.href || '\u2014') : '\u2014',
      });
    }
  }

  // Approach 3: Intercept fetch API for paginated data
  if (notifications.length === 0) {
    console.log('No notifications found via DOM. Trying API...');
    try {
      const res = await fetch('/api/v2/notifications?page=1&per_page=100', { credentials: 'include' });
      const json = await res.json();
      const items = json.data?.notifications || json.notifications || [];
      for (const n of items) {
        const kind = (n.type || '').includes('follow') ? 'follow' : 'like';
        notifications.push({
          kind,
          who: n.sender_nickname || n.sender_name || '',
          who_url: 'https://note.com/' + (n.sender_urlname || ''),
          noticed_at_iso: n.created_at || '',
          noticed_at_jst: '',
          article_title: n.note_title || (kind === 'follow' ? '\u2014' : ''),
          article_url: n.note_url || '\u2014',
          liker_profile: kind === 'follow' ? ('https://note.com/' + (n.sender_urlname || '')) : '\u2014',
        });
      }
    } catch (e) { console.warn('API fetch failed', e); }
  }

  const output = { notifications, extracted_at: new Date().toISOString(), source: 'notifications' };
  const json = JSON.stringify(output, null, 2);
  await navigator.clipboard.writeText(json);
  console.log('Extracted ' + notifications.length + ' notifications. Copied to clipboard.');
  console.log('Paste into the Note.com Analytics app (Data tab).');
})();

2. Profile & Articles

Open your profile page (note.com/your_username), then run: 

// Note.com Analytics -Profile & Articles Extractor
// 1. Open your profile: https://note.com/YOUR_USERNAME
// 2. Scroll down to load all articles
// 3. Paste this entire script into the browser console
// 4. JSON is copied to clipboard -paste into the app

(async () => {
  const articles = [];
  let followerCount = 0;
  let followingCount = 0;
  let noteCount = 0;
  let profileName = '';

  // Approach 1: __NEXT_DATA__ (most reliable)
  const nd = document.getElementById('__NEXT_DATA__');
  if (nd) {
    try {
      const data = JSON.parse(nd.textContent);
      const pp = data?.props?.pageProps;
      const user = pp?.userDetail || pp?.creator || pp?.user || {};

      profileName = user.nickname || user.name || '';
      followerCount = user.followerCount || 0;
      followingCount = user.followingCount || 0;
      noteCount = user.noteCount || 0;

      // Articles may be in contents, notes, or timeline
      const notes = pp?.contents || pp?.notes || pp?.timeline?.contents || [];
      for (const n of notes) {
        const tags = (n.hashtags || n.tags || []).map(
          t => typeof t === 'string' ? t : (t.hashtag?.name || t.name || '')
        ).filter(Boolean);

        articles.push({
          title_jp: n.name || n.title || '',
          title_en: '',
          article_url: 'https://note.com/' + (user.urlname || '') + '/n/' + (n.key || ''),
          published_date_jst: n.publishAt || n.createdAt || '',
          article_type: n.type || n.noteType || '',
          tags: tags,
          views: n.readCount || n.viewCount || 0,
          likes_total: n.likeCount || n.spCount || 0,
        });
      }
    } catch (e) { console.warn('__NEXT_DATA__ parse failed', e); }
  }

  // Approach 2: DOM fallback
  if (articles.length === 0) {
    const cards = document.querySelectorAll(
      '[class*="NoteCard"], [class*="note-card"], article, [class*="Timeline"] a[href*="/n/"]'
    );

    for (const card of cards) {
      const titleEl = card.querySelector('h3, h2, [class*="title"], [class*="Title"]');
      const linkEl = card.querySelector('a[href*="/n/"]') || card.closest('a[href*="/n/"]');
      const timeEl = card.querySelector('time, [datetime]');
      const likeEl = card.querySelector('[class*="like"], [class*="suki"], [class*="spCount"]');
      const tagEls = card.querySelectorAll('[class*="tag"], [class*="hashtag"]');

      if (!titleEl && !linkEl) continue;

      articles.push({
        title_jp: titleEl?.textContent?.trim() || '',
        title_en: '',
        article_url: linkEl?.href?.split('?')[0] || '',
        published_date_jst: timeEl?.getAttribute('datetime') || timeEl?.textContent?.trim() || '',
        article_type: '',
        tags: [...tagEls].map(t => t.textContent?.trim().replace(/^#/, '')).filter(Boolean),
        views: 0,
        likes_total: parseInt(likeEl?.textContent?.replace(/\D/g, '') || '0'),
      });
    }

    // Follower count from DOM
    const followerEl = document.querySelector('[class*="follower"] [class*="count"], [class*="Follower"]');
    if (followerEl) followerCount = parseInt(followerEl.textContent?.replace(/\D/g, '') || '0');
  }

  // Approach 3: API fallback
  if (articles.length === 0) {
    const urlname = location.pathname.replace(/^\//, '').split('/')[0];
    if (urlname) {
      try {
        const res = await fetch('/api/v2/creators/' + urlname + '/contents?kind=note&page=1&per_page=50', { credentials: 'include' });
        const json = await res.json();
        const notes = json.data?.contents || [];
        for (const n of notes) {
          articles.push({
            title_jp: n.name || '',
            title_en: '',
            article_url: 'https://note.com/' + urlname + '/n/' + (n.key || ''),
            published_date_jst: n.publishAt || '',
            article_type: n.type || '',
            tags: (n.hashtags || []).map(t => t.hashtag?.name || t).filter(Boolean),
            views: n.readCount || 0,
            likes_total: n.likeCount || 0,
          });
        }
      } catch (e) { console.warn('API fallback failed', e); }
    }
  }

  const output = {
    articles,
    profile: { name: profileName, followerCount, followingCount, noteCount },
    extracted_at: new Date().toISOString(),
    source: 'profile',
  };
  const json = JSON.stringify(output, null, 2);
  await navigator.clipboard.writeText(json);
  console.log('Extracted ' + articles.length + ' articles. Followers: ' + followerCount);
  console.log('Paste into the Note.com Analytics app (Data tab).');
})();

3. Creator Dashboard

Open your creator dashboard (note.com/dashboard), then run: 

// Note.com Analytics -Creator Dashboard Extractor
// 1. Open https://note.com/dashboard (must be logged in)
// 2. Navigate to the analytics/stats section if available
// 3. Paste this entire script into the browser console
// 4. JSON is copied to clipboard -paste into the app
//
// NOTE: This script attempts to extract whatever analytics
// data note.com exposes. The dashboard structure may vary.

(async () => {
  const stats = { daily_views: [], article_stats: [] };

  // Approach 1: __NEXT_DATA__
  const nd = document.getElementById('__NEXT_DATA__');
  if (nd) {
    try {
      const data = JSON.parse(nd.textContent);
      const pp = data?.props?.pageProps;

      // Look for analytics/stats data in any likely property
      const candidates = [
        pp?.stats, pp?.analytics, pp?.dashboard,
        pp?.dailyStats, pp?.noteStats, pp?.viewStats
      ].filter(Boolean);

      for (const c of candidates) {
        if (Array.isArray(c)) {
          for (const item of c) {
            if (item.date && (item.readCount || item.viewCount)) {
              stats.daily_views.push({
                date: item.date,
                views: item.readCount || item.viewCount || 0,
                likes: item.likeCount || 0,
              });
            }
          }
        }
      }

      // Per-article stats
      const notes = pp?.contents || pp?.notes || [];
      for (const n of notes) {
        stats.article_stats.push({
          title_jp: n.name || n.title || '',
          article_url: n.noteUrl || n.key ? ('https://note.com/n/' + n.key) : '',
          views: n.readCount || n.viewCount || 0,
          likes: n.likeCount || n.spCount || 0,
          comments: n.commentCount || 0,
        });
      }
    } catch (e) { console.warn('__NEXT_DATA__ parse failed', e); }
  }

  // Approach 2: API for stats
  if (stats.daily_views.length === 0 && stats.article_stats.length === 0) {
    try {
      const res = await fetch('/api/v1/stats/pv', { credentials: 'include' });
      if (res.ok) {
        const json = await res.json();
        const daily = json.data || json.stats || [];
        for (const d of daily) {
          stats.daily_views.push({
            date: d.date || d.day || '',
            views: d.pv || d.count || 0,
            likes: d.likes || 0,
          });
        }
      }
    } catch (e) { console.warn('Stats API unavailable', e); }

    try {
      const res = await fetch('/api/v1/stats/notes', { credentials: 'include' });
      if (res.ok) {
        const json = await res.json();
        const notes = json.data || json.notes || [];
        for (const n of notes) {
          stats.article_stats.push({
            title_jp: n.name || '',
            article_url: n.noteUrl || '',
            views: n.readCount || 0,
            likes: n.likeCount || 0,
            comments: n.commentCount || 0,
          });
        }
      }
    } catch (e) { console.warn('Note stats API unavailable', e); }
  }

  const output = {
    stats,
    extracted_at: new Date().toISOString(),
    source: 'dashboard',
  };
  const json = JSON.stringify(output, null, 2);
  await navigator.clipboard.writeText(json);
  const total = stats.daily_views.length + stats.article_stats.length;
  console.log('Extracted ' + stats.daily_views.length + ' daily records, ' + stats.article_stats.length + ' article stats.');
  if (total === 0) {
    console.log('No stats found. The dashboard structure may have changed.');
    console.log('Try navigating to the analytics section first.');
  }
  console.log('Paste into the Note.com Analytics app (Data tab).');
})();
